```cpp
class Solution {
public:
	int minCostToMoveChips(vector<int>& chips) {
		int even = 0, odd = 0;
		for (int i = 0; i < size(chips); i++) {
			if (chips[i] % 2 == 0) { even++; }
			else { odd++; }
		}
		return min(even, odd);
	}
};
```
